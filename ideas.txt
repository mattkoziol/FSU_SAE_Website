import React from 'react';
import execBoard from '../../data/execBoard.json';
import ProfileCard from '../../components/ProfileCard'; // Import the new ProfileCard
import '../../App.css';
import '../../styles/exec.css';

const ExecutiveBoard = () => {
  // Group members by their positions - simplified approach
  const firstRow = execBoard.slice(0, 4); // President, VP, Treasurer, Philanthropy
  const secondRow = [
    execBoard.find(m => m.title === "Member Educator"),
    execBoard.find(m => m.title === "Risk Management Chair"),
    execBoard.find(m => m.title === "Warden"),
    execBoard.find(m => m.title === "Recorder"),
    execBoard.find(m => m.title === "Brotherhood/Alumni Chair")
  ].filter(Boolean); // Remove any undefined members
  
  const thirdRow = [
    ...execBoard.filter(m => m.title.startsWith("Social Chair")),
    ...execBoard.filter(m => m.title.startsWith("House Manager")),
    ...execBoard.filter(m => m.title.startsWith("Rush Chair"))
  ];

  // No longer need renderCard as we are using ProfileCard directly

  return (
    <div className="executive-board-page">
      <header className="page-header" style={{ backgroundColor: '#582C83' }}>
        <div className="container">
          <h1>Executive Board</h1>
          <p className="lead">Meet The Team</p>
        </div>
      </header>

      <section className="exec-positions">
        <div className="container">
          <h2 className="section-title">Executive Board</h2>
          
          {/* First Row - 4 cards */}
          <div className="exec-row exec-row-4">
            {firstRow.map(member => (
              <ProfileCard
                key={member.name}
                name={member.name}
                title={member.title}
                bio={member.bio}
                linkedin={member.linkedin}
                avatarUrl={member.image}
              />
            ))}
          </div>

          {/* Second Row - 5 cards */}
          <div className="exec-row exec-row-5">
            {secondRow.map(member => (
              <ProfileCard
                key={member.name}
                name={member.name}
                title={member.title}
                bio={member.bio}
                linkedin={member.linkedin}
                avatarUrl={member.image}
              />
            ))}
          </div>

          {/* Third Row - 6 cards */}
          <div className="exec-row exec-row-6">
            {thirdRow.map(member => (
              <ProfileCard
                key={member.name}
                name={member.name}
                title={member.title}
                bio={member.bio}
                linkedin={member.linkedin}
                avatarUrl={member.image}
              />
            ))}
          </div>
        </div>
      </section>
    </div>
  );
};

export default ExecutiveBoard;
